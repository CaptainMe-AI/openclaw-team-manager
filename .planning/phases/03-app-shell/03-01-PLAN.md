```typescript
import { create } from "zustand";

interface UIState {
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  closeSidebar: () => void;
}

export const useUIStore = create<UIState>((set) => ({
  sidebarOpen: false,
  toggleSidebar: () => set((state) => ({ sidebarOpen: !state.sidebarOpen })),
  closeSidebar: () => set({ sidebarOpen: false }),
}));
```

CSRF token helper for Devise sign-out. Per RESEARCH.md Pattern 5. Read token fresh from meta tag on each call (not cached):
```typescript
export function getCsrfToken(): string {
  const meta = document.querySelector('meta[name="csrf-token"]');
  return meta?.getAttribute("content") ?? "";
}
```

**Step 4: Create `source/dashboard/app/frontend/router.tsx`.**
Use `createBrowserRouter` with a root layout route. Import page components from `@/components/pages/`. Per D-12, D-13. The root element is a temporary `<Outlet />` wrapper (AppShell is built in Plan 02). Dashboard route uses `index: true`:
```typescript
import { createBrowserRouter, Outlet } from "react-router";
import { DashboardPage } from "@/components/pages/DashboardPage";
import { AgentsPage } from "@/components/pages/AgentsPage";
import { TasksPage } from "@/components/pages/TasksPage";
import { UsagePage } from "@/components/pages/UsagePage";
import { ApprovalsPage } from "@/components/pages/ApprovalsPage";
import { SettingsPage } from "@/components/pages/SettingsPage";

function RootLayout() {
  return (
    <div className="min-h-screen bg-background">
      <Outlet />
    </div>
  );
}

export const router = createBrowserRouter([
  {
    path: "/",
    element: <RootLayout />,
    children: [
      { index: true, element: <DashboardPage /> },
      { path: "agents", element: <AgentsPage /> },
      { path: "tasks", element: <TasksPage /> },
      { path: "usage", element: <UsagePage /> },
      { path: "approvals", element: <ApprovalsPage /> },
      { path: "settings", element: <SettingsPage /> },
    ],
  },
]);
```
IMPORTANT: The `RootLayout` is a temporary wrapper. Plan 02 replaces it with the `AppShell` component. This allows routing to work before the sidebar/topbar are built.

**Step 6: Update `source/dashboard/config/routes.rb`.**
Add a catch-all route AFTER all existing routes (but before the final `end`). Per RESEARCH.md Pitfall 1. The constraint excludes Rails internal routes (`/rails/`) so Active Storage etc. still work:
```ruby
# frozen_string_literal: true

Rails.application.routes.draw do
  devise_for :users, skip: [:registrations]

  # Health check endpoint (Rails 8 default)
  get "up" => "rails/health#show", as: :rails_health_check

  # SPA root -- serves the React app for authenticated users
  root "pages#app"

  # Catch-all for client-side routing (MUST be last)
  get "*path", to: "pages#app", constraints: ->(req) {
    !req.path.start_with?("/rails/")
  }
end
```

**Step 1: Create `source/dashboard/app/frontend/components/ui/PlaceholderSkeleton.tsx`.**
Per D-10, D-11, and UI-SPEC Placeholder Skeleton section. Uses `animate-pulse` for shimmer. Layout is 4 KPI card skeletons in responsive grid + 1 table skeleton below:
```typescript
import { cn } from "@/lib/utils";

interface PlaceholderSkeletonProps {
  className?: string;
}

export function PlaceholderSkeleton({ className }: PlaceholderSkeletonProps) {
  return (
    <div className={cn("space-y-4", className)}>
      {/* KPI card row skeleton */}
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
        {Array.from({ length: 4 }).map((_, i) => (
          <div key={i} className="bg-surface rounded-lg border border-border p-6 animate-pulse">
            <div className="h-3 w-20 bg-border rounded mb-3" />
            <div className="h-6 w-16 bg-border rounded" />
          </div>
        ))}
      </div>
      {/* Table skeleton */}
      <div className="bg-surface rounded-lg border border-border p-6 animate-pulse">
        <div className="h-3 w-32 bg-border rounded mb-4" />
        <div className="space-y-3">
          {Array.from({ length: 5 }).map((_, i) => (
            <div key={i} className="h-3 bg-border rounded" style={{ width: `${80 - i * 8}%` }} />
          ))}
        </div>
      </div>
    </div>
  );
}
```
