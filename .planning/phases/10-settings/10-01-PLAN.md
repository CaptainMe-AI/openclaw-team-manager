    export interface DisplayNameValue { name: string; }
    export interface TimezoneValue { timezone: string; }

    export interface DefaultBudgetValue { amount: number; }

    export interface AuthTokenValue { token: string; }
    export interface SessionPathValue { path: string; }
    export interface DatasourceRefreshValue { seconds: number; }

    // Form state: key -> value mapping for all settings
    export type SettingsFormState = Record<string, unknown>;

    // Helper to extract typed value from a setting
    export function getSettingValue<T>(settings: SettingsFormState, key: string, defaultValue: T): T {
      const val = settings[key];
      if (val === undefined || val === null) return defaultValue;
      return val as T;
